# Cortex Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone MCP server that gives AI agents persistent working memory — tracking what was done, what was learned, what the user intends, and what decisions were made.

**Architecture:** Hybrid capture (zero-cost hooks for events + AI-authored notes for insights) with three-level consolidation (rule-based within session, AI-assisted across sessions, project-level merge). Layered retrieval: always-loaded ~150-200 token header via startup hook, plus on-demand MCP tools. SQLite per project, independent from Nexus.

**Tech Stack:** TypeScript (strict, ES modules), SQLite (better-sqlite3, WAL mode), MCP SDK (@modelcontextprotocol/sdk), Commander (CLI), Vitest (tests)

**Spec:** `docs/superpowers/specs/2026-04-11-cortex-design.md`

```ts
import Database from 'better-sqlite3';

export const SCHEMA_VERSION = 1;

const TABLES = `
CREATE TABLE IF NOT EXISTS meta (
  key   TEXT PRIMARY KEY,
  value TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS sessions (
  id                TEXT PRIMARY KEY,
  parent_session_id TEXT REFERENCES sessions(id) ON DELETE SET NULL,
  started_at        TEXT NOT NULL,
  ended_at          TEXT,
  focus             TEXT,
  agent_type        TEXT NOT NULL DEFAULT 'primary',
  status            TEXT NOT NULL DEFAULT 'active'
);

CREATE TABLE IF NOT EXISTS events (
  id            TEXT PRIMARY KEY,
  session_id    TEXT NOT NULL REFERENCES sessions(id) ON DELETE CASCADE,
  timestamp     TEXT NOT NULL,
  type          TEXT NOT NULL,
  target        TEXT,
  metadata_json TEXT
);

CREATE TABLE IF NOT EXISTS notes (
  id           TEXT PRIMARY KEY,
  session_id   TEXT NOT NULL REFERENCES sessions(id) ON DELETE CASCADE,
  timestamp    TEXT NOT NULL,
  kind         TEXT NOT NULL,
  subject      TEXT,
  content      TEXT NOT NULL,
  alternatives TEXT,
  status       TEXT NOT NULL DEFAULT 'active',
  conflict     INTEGER NOT NULL DEFAULT 0
);

CREATE TABLE IF NOT EXISTS state (
  id         TEXT PRIMARY KEY,
  session_id TEXT REFERENCES sessions(id) ON DELETE CASCADE,
  layer      TEXT NOT NULL,
  content    TEXT NOT NULL,
  created_at TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS token_ledger (
  id         TEXT PRIMARY KEY,
  session_id TEXT NOT NULL REFERENCES sessions(id) ON DELETE CASCADE,
  type       TEXT NOT NULL,
  direction  TEXT NOT NULL,
  tokens     INTEGER NOT NULL,
  timestamp  TEXT NOT NULL
);
`;

const INDEXES = `
CREATE INDEX IF NOT EXISTS idx_events_session       ON events(session_id);
CREATE INDEX IF NOT EXISTS idx_events_type           ON events(session_id, type);
CREATE INDEX IF NOT EXISTS idx_events_timestamp      ON events(timestamp);
CREATE INDEX IF NOT EXISTS idx_notes_session          ON notes(session_id);
CREATE INDEX IF NOT EXISTS idx_notes_kind_subject     ON notes(kind, subject);
CREATE INDEX IF NOT EXISTS idx_notes_status           ON notes(status);
CREATE INDEX IF NOT EXISTS idx_state_session          ON state(session_id);
CREATE INDEX IF NOT EXISTS idx_state_layer            ON state(layer);
CREATE INDEX IF NOT EXISTS idx_ledger_session          ON token_ledger(session_id);
CREATE INDEX IF NOT EXISTS idx_sessions_parent        ON sessions(parent_session_id);
CREATE INDEX IF NOT EXISTS idx_sessions_status        ON sessions(status);
`;

export function openDatabase(dbPath: string): Database.Database {
  const db = new Database(dbPath);

  db.pragma('journal_mode = WAL');
  db.pragma('foreign_keys = ON');
  db.pragma('synchronous = NORMAL');
  db.pragma('busy_timeout = 5000');

  return db;
}

export function applySchema(db: Database.Database): void {
  db.exec(TABLES);
  db.exec(INDEXES);
}

export function initializeMeta(
  db: Database.Database,
  rootPath: string,
): void {
  const upsert = db.prepare(
    'INSERT OR REPLACE INTO meta (key, value) VALUES (?, ?)',
  );

  const initTransaction = db.transaction(() => {
    upsert.run('schema_version', String(SCHEMA_VERSION));
    upsert.run('root_path', rootPath);
    upsert.run('created_at', new Date().toISOString());
  });

  initTransaction();
}
```

```ts
import type Database from 'better-sqlite3';
import { randomUUID } from 'node:crypto';

// ── Row types ────────────────────────────────────────────────────────

export interface SessionRow {
  id: string;
  parent_session_id: string | null;
  started_at: string;
  ended_at: string | null;
  focus: string | null;
  agent_type: string;
  status: string;
}

export interface EventRow {
  id: string;
  session_id: string;
  timestamp: string;
  type: string;
  target: string | null;
  metadata_json: string | null;
}

export interface ParsedEvent {
  id: string;
  session_id: string;
  timestamp: string;
  type: string;
  target: string | null;
  metadata: Record<string, unknown>;
}

// ── Input types ──────────────────────────────────────────────────────

export interface CreateSessionOpts {
  parentSessionId?: string;
  agentType?: string;
  focus?: string;
}

export interface InsertEventOpts {
  sessionId: string;
  type: string;
  target?: string;
  metadata?: Record<string, unknown>;
}

// ── Store ────────────────────────────────────────────────────────────

export class CortexStore {
  constructor(private db: Database.Database) {}

  // ── Sessions ─────────────────────────────────────────────────────

  createSession(opts?: CreateSessionOpts): SessionRow {
    const id = randomUUID();
    const now = new Date().toISOString();
    this.db.prepare(`
      INSERT INTO sessions (id, parent_session_id, started_at, agent_type, focus, status)
      VALUES (?, ?, ?, ?, ?, 'active')
    `).run(
      id,
      opts?.parentSessionId ?? null,
      now,
      opts?.agentType ?? 'primary',
      opts?.focus ?? null,
    );

    return this.getSession(id)!;
  }

  getSession(id: string): SessionRow | undefined {
    return this.db.prepare('SELECT * FROM sessions WHERE id = ?').get(id) as
      | SessionRow
      | undefined;
  }

  getCurrentSession(): SessionRow | undefined {
    return this.db
      .prepare(
        "SELECT * FROM sessions WHERE status = 'active' ORDER BY started_at DESC LIMIT 1",
      )
      .get() as SessionRow | undefined;
  }

  updateSessionFocus(id: string, focus: string): void {
    this.db
      .prepare('UPDATE sessions SET focus = ? WHERE id = ?')
      .run(focus, id);
  }

  endSession(id: string): void {
    this.db
      .prepare(
        "UPDATE sessions SET status = 'ended', ended_at = ? WHERE id = ?",
      )
      .run(new Date().toISOString(), id);
  }

  getRecentSessions(limit: number): SessionRow[] {
    return this.db
      .prepare('SELECT * FROM sessions ORDER BY started_at DESC LIMIT ?')
      .all(limit) as SessionRow[];
  }

  getUnconsolidatedSessions(): SessionRow[] {
    return this.db
      .prepare(`
        SELECT s.* FROM sessions s
        WHERE s.status = 'ended'
          AND NOT EXISTS (
            SELECT 1 FROM state st WHERE st.session_id = s.id AND st.layer = 'session'
          )
        ORDER BY s.started_at ASC
      `)
      .all() as SessionRow[];
  }

  getChildSessions(parentId: string): SessionRow[] {
    return this.db
      .prepare('SELECT * FROM sessions WHERE parent_session_id = ? ORDER BY started_at ASC')
      .all(parentId) as SessionRow[];
  }

  getSessionCount(): number {
    return (
      this.db.prepare('SELECT COUNT(*) as count FROM sessions').get() as {
        count: number;
      }
    ).count;
  }

  // ── Events ───────────────────────────────────────────────────────

  insertEvent(opts: InsertEventOpts): void {
    const id = randomUUID();
    const now = new Date().toISOString();
    this.db.prepare(`
      INSERT INTO events (id, session_id, timestamp, type, target, metadata_json)
      VALUES (?, ?, ?, ?, ?, ?)
    `).run(
      id,
      opts.sessionId,
      now,
      opts.type,
      opts.target ?? null,
      opts.metadata ? JSON.stringify(opts.metadata) : null,
    );
  }

  getEventsBySession(sessionId: string): ParsedEvent[] {
    const rows = this.db
      .prepare('SELECT * FROM events WHERE session_id = ? ORDER BY timestamp ASC')
      .all(sessionId) as EventRow[];
    return rows.map(parseEventRow);
  }

  getEventsByType(sessionId: string, type: string): ParsedEvent[] {
    const rows = this.db
      .prepare(
        'SELECT * FROM events WHERE session_id = ? AND type = ? ORDER BY timestamp ASC',
      )
      .all(sessionId, type) as EventRow[];
    return rows.map(parseEventRow);
  }

  getEventCount(sessionId: string): number {
    return (
      this.db
        .prepare('SELECT COUNT(*) as count FROM events WHERE session_id = ?')
        .get(sessionId) as { count: number }
    ).count;
  }

  deleteEventsBySession(sessionId: string): void {
    this.db
      .prepare('DELETE FROM events WHERE session_id = ?')
      .run(sessionId);
  }

  // ── Utility ──────────────────────────────────────────────────────

  runInTransaction<T>(fn: () => T): T {
    return this.db.transaction(fn)();
  }

  getMeta(key: string): string | undefined {
    return this.db
      .prepare('SELECT value FROM meta WHERE key = ?')
      .pluck()
      .get(key) as string | undefined;
  }

  setMeta(key: string, value: string): void {
    this.db
      .prepare('INSERT OR REPLACE INTO meta (key, value) VALUES (?, ?)')
      .run(key, value);
  }
}

// ── Helpers ──────────────────────────────────────────────────────────

function parseEventRow(row: EventRow): ParsedEvent {
  return {
    id: row.id,
    session_id: row.session_id,
    timestamp: row.timestamp,
    type: row.type,
    target: row.target,
    metadata: row.metadata_json ? JSON.parse(row.metadata_json) : {},
  };
}
```

```ts
  // ── Notes ────────────────────────────────────────────────────────

  private static readonly SUBJECT_REQUIRED_KINDS = new Set([
    'decision', 'intent', 'blocker', 'focus',
  ]);

  private static readonly AUTO_SUPERSEDE_KINDS = new Set([
    'decision', 'intent',
  ]);

  insertNote(opts: InsertNoteOpts): ParsedNote {
    const { sessionId, kind, content, alternatives } = opts;
    let subject = opts.subject ?? null;

    // Validate subject requirement
    if (CortexStore.SUBJECT_REQUIRED_KINDS.has(kind) && !subject) {
      throw new Error(`Subject is required for '${kind}' notes`);
    }

    // Normalize subject
    if (subject) {
      subject = subject.trim().toLowerCase();
    }

    // Auto-supersede prior active notes with same kind + subject
    if (subject && CortexStore.AUTO_SUPERSEDE_KINDS.has(kind)) {
      this.db.prepare(`
        UPDATE notes SET status = 'superseded'
        WHERE kind = ? AND subject = ? AND status = 'active'
      `).run(kind, subject);
    }

    const id = randomUUID();
    const now = new Date().toISOString();

    this.db.prepare(`
      INSERT INTO notes (id, session_id, timestamp, kind, subject, content, alternatives, status, conflict)
      VALUES (?, ?, ?, ?, ?, ?, ?, 'active', 0)
    `).run(
      id,
      sessionId,
      now,
      kind,
      subject,
      content,
      alternatives ? JSON.stringify(alternatives) : null,
    );

    // Focus note: update session focus directly
    if (kind === 'focus' && subject) {
      this.updateSessionFocus(sessionId, subject);
    }

    // First intent note: set session focus if none set
    if (kind === 'intent' && subject) {
      const session = this.getSession(sessionId);
      if (session && !session.focus) {
        this.updateSessionFocus(sessionId, subject);
      }
    }

    return this.getNote(id)!;
  }

  getNote(id: string): ParsedNote | undefined {
    const row = this.db.prepare('SELECT * FROM notes WHERE id = ?').get(id) as
      | NoteRow
      | undefined;
    return row ? parseNoteRow(row) : undefined;
  }

  getNotesBySession(sessionId: string): ParsedNote[] {
    const rows = this.db
      .prepare('SELECT * FROM notes WHERE session_id = ? ORDER BY timestamp ASC')
      .all(sessionId) as NoteRow[];
    return rows.map(parseNoteRow);
  }

  getActiveNotes(sessionId?: string): ParsedNote[] {
    if (sessionId) {
      const rows = this.db
        .prepare(
          "SELECT * FROM notes WHERE session_id = ? AND status = 'active' ORDER BY timestamp ASC",
        )
        .all(sessionId) as NoteRow[];
      return rows.map(parseNoteRow);
    }

    const rows = this.db
      .prepare(
        "SELECT * FROM notes WHERE status = 'active' ORDER BY timestamp ASC",
      )
      .all() as NoteRow[];
    return rows.map(parseNoteRow);
  }

  getNotesByKindAndSubject(kind: string, subject: string): ParsedNote[] {
    const normalized = subject.trim().toLowerCase();
    const rows = this.db
      .prepare(
        'SELECT * FROM notes WHERE kind = ? AND subject = ? ORDER BY timestamp ASC',
      )
      .all(kind, normalized) as NoteRow[];
    return rows.map(parseNoteRow);
  }

  updateNoteStatus(id: string, status: 'active' | 'superseded' | 'resolved'): void {
    this.db.prepare('UPDATE notes SET status = ? WHERE id = ?').run(status, id);
  }

  getNotesByStatus(status: string): ParsedNote[] {
    const rows = this.db
      .prepare('SELECT * FROM notes WHERE status = ? ORDER BY timestamp ASC')
      .all(status) as NoteRow[];
    return rows.map(parseNoteRow);
  }

  markConflict(id: string): void {
    this.db.prepare('UPDATE notes SET conflict = 1 WHERE id = ?').run(id);
  }
```

```ts
  insertState(opts: InsertStateOpts): void {
    const id = randomUUID();
    const now = new Date().toISOString();
    this.db.prepare(`
      INSERT INTO state (id, session_id, layer, content, created_at)
      VALUES (?, ?, ?, ?, ?)
    `).run(id, opts.sessionId ?? null, opts.layer, opts.content, now);

    return this.db
      .prepare(
        "SELECT * FROM state WHERE session_id = ? AND layer = 'session' ORDER BY created_at DESC LIMIT 1",
      )
      .get(sessionId) as StateRow | undefined;
  }

  getProjectState(): StateRow | undefined {
    return this.db
      .prepare(
        "SELECT * FROM state WHERE session_id IS NULL AND layer = 'project' ORDER BY created_at DESC LIMIT 1",
      )
      .get() as StateRow | undefined;
  }

  replaceProjectState(content: string): void {
    this.db.prepare(
      "DELETE FROM state WHERE session_id IS NULL AND layer = 'project'",
    ).run();
    this.insertState({ layer: 'project', content });
  }

  getRecentStates(limit: number): StateRow[] {
    return this.db
      .prepare(
        "SELECT * FROM state WHERE layer = 'session' ORDER BY created_at DESC LIMIT ?",
      )
      .all(limit) as StateRow[];
  }

  // ── Token Ledger ─────────────────────────────────────────────────

  insertLedgerEntry(opts: InsertLedgerOpts): void {
    const id = randomUUID();
    const now = new Date().toISOString();
    this.db.prepare(`
      INSERT INTO token_ledger (id, session_id, type, direction, tokens, timestamp)
      VALUES (?, ?, ?, ?, ?, ?)
    `).run(id, opts.sessionId, opts.type, opts.direction, opts.tokens, now);
  }

  getLedgerBySession(sessionId: string): LedgerRow[] {
    return this.db

  }

  getTotalTokens(): { spent: number; saved: number } {
    const row = this.db
      .prepare(`
        SELECT
          COALESCE(SUM(CASE WHEN direction = 'spent' THEN tokens ELSE 0 END), 0) as spent,
          COALESCE(SUM(CASE WHEN direction = 'saved' THEN tokens ELSE 0 END), 0) as saved
        FROM token_ledger
      `)
      .get() as { spent: number; saved: number };

    saved: number;

        FROM token_ledger
        GROUP BY type, direction

    const byType: Record<string, { spent: number; saved: number }> = {};
    for (const row of rows) {
      if (!byType[row.type]) byType[row.type] = { spent: 0, saved: 0 };
      byType[row.type][row.direction as 'spent' | 'saved'] = row.total;
    }

    return { ...totals, byType };
```

```ts
// ── Command Classification ───────────────────────────────────────────

const SAFE_CATEGORIES: [RegExp, string][] = [
  [/^git\b/, 'git'],
  [/^(?:npm|npx|yarn|pnpm|bun)\b/, 'npm'],
  [/^(?:vitest|jest|pytest|go\s+test|cargo\s+test|dotnet\s+test|mocha|ava)\b/, 'test'],
  [/^(?:tsc|cargo\s+build|go\s+build|make|cmake|gradle|mvn|dotnet\s+build)\b/, 'build'],
  [/^(?:node|python|python3|go\s+run|cargo\s+run|dotnet\s+run)\b/, 'run'],
  [/^(?:ls|dir|cat|head|tail|wc|find|grep|rg|fd|tree)\b/, 'read'],
  [/^(?:cd|pwd|echo|mkdir|cp|mv)\b/, 'shell'],
];

export function classifyCommand(cmd: string): string {
  const trimmed = cmd.trim();
  for (const [pattern, category] of SAFE_CATEGORIES) {
    if (pattern.test(trimmed)) return category;
  }
  return 'other';
}

// ── Secret Redaction ─────────────────────────────────────────────────

const SECRET_PATTERNS: RegExp[] = [
  // API key patterns: sk-*, ghp_*, glpat-*, etc.
  /\b(?:sk-|ghp_|gho_|ghu_|ghs_|ghr_|glpat-|xoxb-|xoxp-|xoxs-|AKIA)[A-Za-z0-9_-]{10,}\b/g,
  // Bearer tokens
  /Bearer\s+[A-Za-z0-9._~+/=-]{20,}/gi,
  // --password=VALUE or --password VALUE
  /(--password[=\s]+)\S+/gi,
  // --token=VALUE or --token VALUE
  /(--token[=\s]+)\S+/gi,
  // --secret=VALUE or --secret VALUE
  /(--secret[=\s]+)\S+/gi,
  // --api-key=VALUE or --api-key VALUE
  /(--api-key[=\s]+)\S+/gi,
  // ENV=secret_value at start of command (KEY=value where value looks like a secret)
  /^([A-Z_]+=)(?:sk-|ghp_|Bearer\s+)[^\s]*/gm,
  // Generic KEY=value with secret-looking keys
  /\b(?:API_KEY|SECRET|TOKEN|PASSWORD|CREDENTIALS|AUTH)=[^\s]+/gi,
];

export function redactCommand(cmd: string): string {
  if (!cmd.trim()) return cmd;

  let result = cmd;
  for (const pattern of SECRET_PATTERNS) {
    // Reset lastIndex for global regexps
    pattern.lastIndex = 0;
    result = result.replace(pattern, (match, capturedPrefix?: string) => {
      // For patterns that capture the flag prefix (--password=), keep the prefix
      if (capturedPrefix) {
        return `${capturedPrefix}[REDACTED]`;
      }
      return '[REDACTED]';
    });
  }

  return result;
}

// ── File Extraction ──────────────────────────────────────────────────

const FILE_PATTERNS: RegExp[] = [
  // Explicit file paths in arguments
  /(?:^|\s)((?:\.{0,2}\/)?[\w./-]+\.(?:ts|tsx|js|jsx|py|go|rs|java|cs|css|json|md|yaml|yml|toml|sql|sh))\b/g,
];

export function extractTouchedFiles(cmd: string): string[] {
  const files = new Set<string>();
  for (const pattern of FILE_PATTERNS) {
    pattern.lastIndex = 0;
    let match;
    while ((match = pattern.exec(cmd)) !== null) {
      files.add(match[1].trim());
    }
  }
  return [...files];
}
```

```ts
import { CortexStore } from '../db/store.js';
import { redactCommand, classifyCommand, extractTouchedFiles } from './redact.js';

// ── Argument Types ───────────────────────────────────────────────────

export interface ReadArgs {
  file: string;
  lines?: string;
}

export interface EditArgs {
  file: string;
  lines?: string;
}

export interface WriteArgs {
  file: string;
}

export interface CmdArgs {
  exit?: string;
  cmd?: string;
}

export interface AgentArgs {
  desc: string;
}

// ── Line Range Parsing ───────────────────────────────────────────────

function parseLineRange(lines?: string): { line_start?: number; line_end?: number } {
  if (!lines) return {};
  const match = lines.match(/^(\d+)-(\d+)$/);
  if (!match) return {};
  return { line_start: parseInt(match[1], 10), line_end: parseInt(match[2], 10) };
}

// ── Handlers ─────────────────────────────────────────────────────────

export function handleReadEvent(
  store: CortexStore,
  sessionId: string,
  args: ReadArgs,
): void {
  const lineRange = parseLineRange(args.lines);
  store.insertEvent({
    sessionId,
    type: 'read',
    target: args.file,
    metadata: {
      ...lineRange,
    },
  });
}

export function handleEditEvent(
  store: CortexStore,
  sessionId: string,
  args: EditArgs,
): void {
  const lineRange = parseLineRange(args.lines);
  store.insertEvent({
    sessionId,
    type: 'edit',
    target: args.file,
    metadata: {
      ...lineRange,
    },
  });
}

export function handleWriteEvent(
  store: CortexStore,
  sessionId: string,
  args: WriteArgs,
): void {
  store.insertEvent({
    sessionId,
    type: 'write',
    target: args.file,
  });
}

export function handleCmdEvent(
  store: CortexStore,
  sessionId: string,
  args: CmdArgs,
): void {
  const exitCode = args.exit !== undefined ? parseInt(args.exit, 10) : undefined;
  const category = args.cmd ? classifyCommand(args.cmd) : undefined;
  const safeSummary = args.cmd ? redactCommand(args.cmd) : undefined;
  const filesTouched = args.cmd ? extractTouchedFiles(args.cmd) : [];

  store.insertEvent({
    sessionId,
    type: 'cmd',
    metadata: {
      ...(exitCode !== undefined ? { exit_code: exitCode } : {}),
      ...(category ? { category } : {}),
      ...(safeSummary ? { safe_summary: safeSummary } : {}),
      ...(filesTouched.length > 0 ? { files_touched: filesTouched } : {}),
    },
  });
}

export function handleAgentEvent(
  store: CortexStore,
  sessionId: string,
  args: AgentArgs,
): void {
  store.insertEvent({
    sessionId,
    type: 'agent',
    metadata: {
      description: args.desc,
    },
  });
}
```

```ts
import { CortexStore, type ParsedEvent } from '../db/store.js';

// ── Compressed event types ───────────────────────────────────────────

export interface CompressedEvent {
  type: string;
  target?: string;
  count?: number;
  line_ranges?: [number, number][];
  iterations?: number;
  files?: string[];
  description?: string;
  exit_code?: number;
  category?: string;
}

// ── Level 1: Rule-based compression ──────────────────────────────────

export function consolidateLevel1(
  store: CortexStore,
  sessionId: string,
): CompressedEvent[] {
  const events = store.getEventsBySession(sessionId);
  if (events.length === 0) return [];

  // Pass 1: Collapse test cycles
  const afterCycles = collapseTestCycles(events);

  // Pass 2: Dedup + merge by type and target
  return deduplicateEvents(afterCycles);
}

// ── Test Cycle Detection ─────────────────────────────────────────────

interface IntermediateEvent {
  type: string;
  original?: ParsedEvent;
  compressed?: CompressedEvent;
}

function collapseTestCycles(events: ParsedEvent[]): IntermediateEvent[] {
  const result: IntermediateEvent[] = [];
  let i = 0;

  while (i < events.length) {
    // Look for test fail (cmd with exit_code != 0 and category = test)
    if (isTestFail(events[i])) {
      const cycle = tryCollapseCycle(events, i);
      if (cycle) {
        result.push({ type: 'test_cycle', compressed: cycle.compressed });
        i = cycle.endIndex + 1;
        continue;
      }
    }
    result.push({ type: events[i].type, original: events[i] });
    i++;
  }

  return result;
}

function isTestFail(event: ParsedEvent): boolean {
  return (
    event.type === 'cmd' &&
    event.metadata.category === 'test' &&
    event.metadata.exit_code !== undefined &&
    event.metadata.exit_code !== 0
  );
}

function isTestPass(event: ParsedEvent): boolean {
  return (
    event.type === 'cmd' &&
    event.metadata.category === 'test' &&
    event.metadata.exit_code === 0
  );
}

function tryCollapseCycle(
  events: ParsedEvent[],
  startIndex: number,
): { compressed: CompressedEvent; endIndex: number } | null {
  let iterations = 0;
  const filesEdited = new Set<string>();
  let j = startIndex;

  while (j < events.length) {
    // Expect: test fail
    if (!isTestFail(events[j])) break;
    j++;
    iterations++;

    // Consume edits until next test command
    while (j < events.length && events[j].type !== 'cmd') {
      if (events[j].type === 'edit' && events[j].target) {
        filesEdited.add(events[j].target!);
      }
      j++;
    }

    // Check if next cmd is a test pass
    if (j < events.length && isTestPass(events[j])) {
      return {
        compressed: {
          type: 'test_cycle',
          iterations,
          files: [...filesEdited],
        },
        endIndex: j,
      };
    }

    // If next cmd is another test fail, continue the loop
    // If it's not a test at all, this isn't a cycle
    if (j >= events.length || events[j].metadata.category !== 'test') {
      return null;
    }
  }

  return null;
}

// ── Deduplication ────────────────────────────────────────────────────

function deduplicateEvents(events: IntermediateEvent[]): CompressedEvent[] {
  // Group file-based events by type + target
  const fileGroups = new Map<string, IntermediateEvent[]>();
  const result: CompressedEvent[] = [];

  for (const event of events) {
    // Pre-compressed events (test cycles) pass through directly
    if (event.compressed) {
      result.push(event.compressed);
      continue;
    }

    const original = event.original!;

    // Agent and non-file events pass through
    if (original.type === 'agent') {
      result.push({
        type: 'agent',
        description: original.metadata.description as string,
      });
      continue;
    }

    if (original.type === 'cmd') {
      // Non-test commands are kept individually
      result.push({
        type: 'cmd',
        ...(original.metadata.exit_code !== undefined ? { exit_code: original.metadata.exit_code as number } : {}),
        ...(original.metadata.category ? { category: original.metadata.category as string } : {}),
      });
      continue;
    }

    // Group reads, edits, writes by target
    if (original.target) {
      const key = `${original.type}:${original.target}`;
      if (!fileGroups.has(key)) fileGroups.set(key, []);
      fileGroups.get(key)!.push(event);
    } else {
      result.push({ type: original.type, count: 1 });
    }
  }

  // Merge file groups
  for (const [key, group] of fileGroups) {
    const [type, target] = key.split(':', 2);
    const lineRanges: [number, number][] = [];

    for (const event of group) {
      const meta = event.original!.metadata;
      if (meta.line_start !== undefined && meta.line_end !== undefined) {
        const range: [number, number] = [meta.line_start as number, meta.line_end as number];
        // Dedup identical ranges
        if (!lineRanges.some(([s, e]) => s === range[0] && e === range[1])) {
          lineRanges.push(range);
        }
      }
    }

    result.push({
      type,
      target,
      count: group.length,
      line_ranges: lineRanges,
    });
  }

  return result;
}

// ── Render compressed events to text ─────────────────────────────────

export function renderCompressed(events: CompressedEvent[]): string {
  const lines: string[] = [];

  for (const event of events) {
    switch (event.type) {
      case 'read': {
        const ranges = event.line_ranges?.length
          ? ` (lines: ${event.line_ranges.map(([s, e]) => `${s}-${e}`).join(', ')})`
          : '';
        lines.push(`Read ${event.target} x${event.count}${ranges}`);
        break;
      }
      case 'edit': {
        const ranges = event.line_ranges?.length
          ? ` (lines: ${event.line_ranges.map(([s, e]) => `${s}-${e}`).join(', ')})`
          : '';
        lines.push(`Edited ${event.target} x${event.count}${ranges}`);
        break;
      }
      case 'write':
        lines.push(`Created ${event.target}`);
        break;
      case 'test_cycle':
        lines.push(
          `Test cycle: fixed after ${event.iterations} iteration${event.iterations === 1 ? '' : 's'}` +
            (event.files?.length ? ` (${event.files.join(', ')})` : ''),
        );
        break;
      case 'cmd':
        lines.push(
          `Command (${event.category ?? 'other'}): exit ${event.exit_code ?? '?'}`,
        );
        break;
      case 'agent':
        lines.push(`Subagent: ${event.description}`);
        break;
      default:
        lines.push(`${event.type} x${event.count ?? 1}`);
    }
  }

  return lines.join('\n');
}
```

```ts
// ── Header Generation ────────────────────────────────────────────────

export function buildHeader(store: CortexStore): string {
  const sessions = store.getRecentSessions(10);
  if (sessions.length === 0) {
    return 'Cortex: no prior sessions';
  }

  const sessionCount = store.getSessionCount();
  const totals = store.getTotalTokens();
  const savingsStr = totals.saved > 0
    ? ` | ~${formatTokens(totals.saved)} tokens saved`
    : '';

  // Determine focus: most recent session with a focus
  const focusSession = sessions.find((s) => s.focus);
  const focus = focusSession?.focus ?? 'unfocused';

  // Check for project-level state first
  const projectState = store.getProjectState();
  if (projectState) {
    return [
      `Cortex: ${focus} | ${sessionCount} sessions${savingsStr}`,
      projectState.content,
    ].join('\n');
  }

  // Check for consolidated session states
  const lastEnded = sessions.find((s) => s.status === 'ended');
  if (lastEnded) {
    const sessionState = store.getSessionState(lastEnded.id);
    if (sessionState) {
      return [
        `Cortex: ${focus} | ${sessionCount} sessions${savingsStr}`,
        sessionState.content,
      ].join('\n');
    }
  }

  // Provisional header from unconsolidated data
  const unconsolidated = store.getUnconsolidatedSessions();
  if (unconsolidated.length > 0) {
    return buildProvisionalHeader(store, focus, sessionCount, savingsStr, unconsolidated);
  }

  return `Cortex: ${focus} | ${sessionCount} sessions${savingsStr}`;
}

function buildProvisionalHeader(
  store: CortexStore,
  focus: string,
  sessionCount: number,
  savingsStr: string,
  unconsolidated: { id: string }[],
): string {
  const lines: string[] = [
    `Cortex [provisional]: ${focus} | ${sessionCount} sessions${savingsStr}`,
  ];

  // Gather file activity from unconsolidated sessions
  const fileActivity = new Map<string, { reads: number; edits: number }>();
  let cmdCount = 0;
  let noteCount = 0;

  for (const session of unconsolidated) {
    const events = store.getEventsBySession(session.id);
    for (const event of events) {
      if (event.target && (event.type === 'read' || event.type === 'edit')) {
        const entry = fileActivity.get(event.target) ?? { reads: 0, edits: 0 };
        if (event.type === 'read') entry.reads++;
        else entry.edits++;
        fileActivity.set(event.target, entry);
      }
      if (event.type === 'cmd') cmdCount++;
    }

    noteCount += store.getActiveNotes(session.id).length;
  }

  // File summary
  if (fileActivity.size > 0) {
    const fileSummary = [...fileActivity.entries()]
      .sort((a, b) => (b[1].reads + b[1].edits) - (a[1].reads + a[1].edits))
      .slice(0, 5)
      .map(([file, counts]) => {
        const parts: string[] = [];
        if (counts.reads > 0) parts.push(`${counts.reads} reads`);
        if (counts.edits > 0) parts.push(`${counts.edits} edits`);
        return `${file} (${parts.join(', ')})`;
      })
      .join(', ');
    lines.push(`Touched: ${fileSummary}`);
  }

  if (cmdCount > 0) lines.push(`Commands: ${cmdCount}`);
  if (noteCount > 0) lines.push(`Active notes: ${noteCount}`);

  lines.push(`→ Call cortex_state for full briefing`);
  return lines.join('\n');
}

// ── Full State ───────────────────────────────────────────────────────

export function buildFullState(store: CortexStore): string {
  const sections: string[] = [];

  // Active notes grouped by kind
  const activeNotes = store.getActiveNotes();
  const byKind = new Map<string, typeof activeNotes>();
  for (const note of activeNotes) {
    const list = byKind.get(note.kind) ?? [];
    list.push(note);
    byKind.set(note.kind, list);
  }

  const kindOrder = ['intent', 'decision', 'blocker', 'insight'];
  for (const kind of kindOrder) {
    const notes = byKind.get(kind);
    if (!notes || notes.length === 0) continue;

    const kindLabel = kind.charAt(0).toUpperCase() + kind.slice(1);
    const noteLines = notes.map((n) => {
      const subj = n.subject ? `[${n.subject}] ` : '';
      const conflict = n.conflict ? ' ⚠️ conflict' : '';
      return `- ${subj}${n.content}${conflict}`;
    });
    sections.push(`${kindLabel}s:\n${noteLines.join('\n')}`);
  }

  // Recent session activity (compressed events)
  const sessions = store.getRecentSessions(3);
  for (const session of sessions) {
    const compressed = consolidateLevel1(store, session.id);
    if (compressed.length > 0) {
      const rendered = renderCompressed(compressed);
      const focusLabel = session.focus ? ` (${session.focus})` : '';
      sections.push(`Session${focusLabel}:\n${rendered}`);
    }
  }

  // Project state if available
  const projectState = store.getProjectState();
  if (projectState) {
    sections.push(`Project context:\n${projectState.content}`);
  }

  return sections.join('\n\n');
}

// ── Helpers ──────────────────────────────────────────────────────────

function formatTokens(n: number): string {
  if (n >= 1000) return `${(n / 1000).toFixed(1)}k`;
  return String(n);
}
```

```ts
import { CortexStore, type ParsedNote } from '../db/store.js';

export function recall(store: CortexStore, topic: string): string {
  const topicLower = topic.toLowerCase();
  const matches: { note: ParsedNote; relevance: number }[] = [];

  // Search active notes
  const activeNotes = store.getActiveNotes();
  for (const note of activeNotes) {
    const relevance = scoreRelevance(note, topicLower);
    if (relevance > 0) {
      matches.push({ note, relevance });
    }
  }

  // Search resolved notes (lower relevance)
  const resolvedNotes = store.getNotesByStatus('resolved');
  for (const note of resolvedNotes) {
    const relevance = scoreRelevance(note, topicLower) * 0.5;
    if (relevance > 0) {
      matches.push({ note, relevance });
    }
  }

  // Sort by relevance (focus-matching notes get a boost)
  const currentSession = store.getCurrentSession();
  const currentFocus = currentSession?.focus?.toLowerCase();

  matches.sort((a, b) => {
    let aScore = a.relevance;
    let bScore = b.relevance;
    if (currentFocus) {
      if (a.note.subject?.toLowerCase().includes(currentFocus)) aScore += 2;
      if (b.note.subject?.toLowerCase().includes(currentFocus)) bScore += 2;
    }
    return bScore - aScore;
  });

  // Build response
  const lines: string[] = [];

  if (matches.length > 0) {
    for (const { note } of matches.slice(0, 10)) {
      const kindLabel = note.kind.charAt(0).toUpperCase() + note.kind.slice(1);
      const subj = note.subject ? `[${note.subject}] ` : '';
      const status = note.status !== 'active' ? ` (${note.status})` : '';
      lines.push(`${kindLabel}: ${subj}${note.content}${status}`);
    }
  }

  // Also search consolidated state
  const stateMatches: string[] = [];
  const recentStates = store.getRecentStates(5);
  for (const state of recentStates) {
    if (state.content.toLowerCase().includes(topicLower)) {
      stateMatches.push(state.content);
    }
  }
  const projectState = store.getProjectState();
  if (projectState?.content.toLowerCase().includes(topicLower)) {
    stateMatches.push(projectState.content);
  }

  if (stateMatches.length > 0) {
    lines.push('---');
    lines.push(...stateMatches.slice(0, 3));
  }

  if (lines.length === 0) {
    return `No matches for "${topic}".`;
  }

  return lines.join('\n');
}

// ── Relevance Scoring ────────────────────────────────────────────────

function scoreRelevance(note: ParsedNote, topicLower: string): number {
  let score = 0;

  // Subject match (highest value)
  if (note.subject?.toLowerCase().includes(topicLower)) {
    score += 5;
  }

  // Content match
  if (note.content.toLowerCase().includes(topicLower)) {
    score += 3;
  }

  // Kind boost: decisions and intents > insights
  if (note.kind === 'decision' || note.kind === 'intent') score += 1;
  if (note.kind === 'blocker') score += 1;

  return score;
}
```

```ts
import { CortexStore } from '../db/store.js';

export function brief(store: CortexStore, topic: string, forAgent?: string): string {
  const topicLower = topic.toLowerCase();
  const lines: string[] = [];

  if (forAgent) {
    lines.push(`Briefing for ${forAgent}:`);
  }

  // Get relevant active notes
  const activeNotes = store.getActiveNotes();
  const relevant = activeNotes.filter((note) => {
    if (note.subject?.toLowerCase().includes(topicLower)) return true;
    if (note.content.toLowerCase().includes(topicLower)) return true;
    return false;
  });

  // Prioritize: decisions > intents > blockers > insights
  const kindPriority: Record<string, number> = {
    decision: 0,
    intent: 1,
    blocker: 2,
    insight: 3,
  };

  relevant.sort(
    (a, b) => (kindPriority[a.kind] ?? 4) - (kindPriority[b.kind] ?? 4),
  );

  // Keep it concise — max 5 notes
  for (const note of relevant.slice(0, 5)) {
    const subj = note.subject ? `[${note.subject}] ` : '';
    lines.push(`${note.kind}: ${subj}${note.content}`);
  }

  // Add current focus context
  const currentSession = store.getCurrentSession();
  if (currentSession?.focus) {
    lines.push(`Current focus: ${currentSession.focus}`);
  }

  if (lines.length === 0 || (forAgent && lines.length === 1)) {
    lines.push(`No context found for "${topic}".`);
  }

  return lines.join('\n');
}
```

```ts
export function handleToolCall(
  store: CortexStore,
  toolName: string,
  args: Record<string, unknown>,
): string {
  const sessionId = ensureSession(store);

  switch (toolName) {
    case 'cortex_state': {
      return buildFullState(store);
    }

    case 'cortex_note': {
      const kind = args.kind as string;
      const content = args.content as string;
      const subject = args.subject as string | undefined;
      const alternatives = args.alternatives as string[] | undefined;

      try {
        store.insertNote({ sessionId, kind: kind as InsertNoteOpts['kind'], content, subject, alternatives });
        const subjectStr = subject ? ` [${subject}]` : '';
        return `Noted (${kind}${subjectStr}): ${content.slice(0, 60)}${content.length > 60 ? '...' : ''}`;
      } catch (error) {
        const msg = error instanceof Error ? error.message : 'Unknown error';
        return `Error: ${msg}`;
      }
    }

    case 'cortex_recall': {
      const topic = args.topic as string;
      return recall(store, topic);
    }

    case 'cortex_brief': {
      const topic = args.topic as string;
      const forAgent = args.for as string | undefined;
      return brief(store, topic, forAgent);
    }

    default:
      return `Unknown tool: ${toolName}`;
  }
}

  const current = store.getCurrentSession();
  if (current) return current.id;

export function createMcpServer(store: CortexStore): Server {
  const server = new Server(
    { name: 'cortex', version: '0.1.0' },
    { capabilities: { tools: {} } },
  );

  server.setRequestHandler(ListToolsRequestSchema, async () => {
    return { tools: TOOL_DEFINITIONS };
  });

  server.setRequestHandler(CallToolRequestSchema, async (request) => {
    const { name, arguments: args } = request.params;
    const result = handleToolCall(store, name, (args ?? {}) as Record<string, unknown>);
    return { content: [{ type: 'text', text: result }] };
  });

  return server;
}

export async function startServer(startDir?: string): Promise<void> {
  const cwd = startDir ?? process.cwd();
  const dbPath = path.join(cwd, '.cortex.db');
  const isNew = !fs.existsSync(dbPath);

  applySchema(db);
  if (isNew) {
    initializeMeta(db, cwd);
  }
  const store = new CortexStore(db);

  // Ensure a session exists
  ensureSession(store);

  const server = createMcpServer(store);
  const transport = new StdioServerTransport();
  await server.connect(transport);
  console.error('Cortex MCP server running on stdio');
}
```

```ts
export function getPendingConsolidation(store: CortexStore): SessionRow[] {

export function writeSessionSummary(
  store: CortexStore,
  sessionId: string,
  summary: string,
): void {
  store.insertState({
    sessionId,
    layer: 'session',
    content: summary,
  });

  // Prune raw events now that we have a summary
  store.deleteEventsBySession(sessionId);

// ── Subagent Note Promotion ──────────────────────────────────────────

export function promoteSubagentNotes(
  store: CortexStore,
  parentSessionId: string,
): void {
  const children = store.getChildSessions(parentSessionId);
  const parentNotes = store.getActiveNotes(parentSessionId);

  for (const child of children) {
    const childNotes = store.getActiveNotes(child.id);

    for (const childNote of childNotes) {
      // Check for exact duplicate (same kind + subject + content)
      const isDuplicate = parentNotes.some(
        (pn) =>
          pn.kind === childNote.kind &&
          pn.subject === childNote.subject &&
          pn.content === childNote.content,
      );

      if (isDuplicate) continue;

      // Check for conflict (same kind + subject, different content)
      const conflicting = parentNotes.find(
        (pn) =>
          pn.kind === childNote.kind &&
          pn.subject === childNote.subject &&
          pn.subject !== null &&
          pn.content !== childNote.content,
      );

      // Promote: insert copy into parent session
      const promoted = store.insertNote({
        sessionId: parentSessionId,
        kind: childNote.kind as InsertNoteOpts['kind'],
        content: childNote.content,
        ...(childNote.subject ? { subject: childNote.subject } : {}),
        ...(childNote.alternatives ? { alternatives: childNote.alternatives } : {}),
      });

      if (conflicting) {
        // Flag both as conflict
        store.markConflict(conflicting.id);
        store.markConflict(promoted.id);

}
```

```ts
// ── Level 3: Cross-session merge ─────────────────────────────────────

const MERGE_THRESHOLD = 5;

export function mergeProjectState(store: CortexStore): boolean {
  const allStates = store.getRecentStates(100);
  if (allStates.length <= MERGE_THRESHOLD) return false;

  // Keep the most recent MERGE_THRESHOLD session states as-is
  // Merge the older ones into project state
  const recent = allStates.slice(0, MERGE_THRESHOLD);
  const older = allStates.slice(MERGE_THRESHOLD);

  if (older.length === 0) return false;

  // Build merged project state from older session summaries + active notes
  const sections: string[] = [];

  // Collect older session summaries
  const sessionSummaries = older.map((s) => s.content).join('\n');
  sections.push(sessionSummaries);

  // Include active notes summary
  const activeNotes = store.getActiveNotes();
  if (activeNotes.length > 0) {
    const notesSummary = activeNotes
      .slice(0, 20)
      .map((n) => {
        const subj = n.subject ? `[${n.subject}] ` : '';
        return `${n.kind}: ${subj}${n.content}`;
      })
      .join('\n');
    sections.push(notesSummary);
  }

  // Existing project state gets folded in
  const existingProject = store.getProjectState();
  if (existingProject) {
    sections.unshift(existingProject.content);
  }

  const mergedContent = sections.join('\n\n');

  // Replace project state (ceiling: truncate to ~500 tokens worth, ~2000 chars)
  const truncated = mergedContent.length > 2000
    ? mergedContent.slice(0, 2000) + '\n[truncated]'
    : mergedContent;

  store.replaceProjectState(truncated);

  return true;
}
```
